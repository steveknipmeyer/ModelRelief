# Default server configuration
#
server {
    listen         80;
    server_name    modelrelief.org www.modelrelief.org;
    return         301 https://modelrelief.org$request_uri;
    }

server {
    listen 443 ssl;
    server_name    modelrelief.org www.modelrelief.org;

    ssl_certificate /root/.acme.sh/modelrelief.org/fullchain.cer;
    ssl_certificate_key /root/.acme.sh/modelrelief.org/modelrelief.org.key;

    location / {
        proxy_pass         https://localhost:5001;
        proxy_http_version 1.1;
        proxy_set_header   Upgrade $http_upgrade;
        proxy_set_header   Connection keep-alive;
        proxy_set_header   Host $host;
        proxy_cache_bypass $http_upgrade;
        proxy_set_header   X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header   X-Forwarded-Proto $scheme;
    }
}
